factory PeripheralVersion.fromInt32(int v) {
  int major = (v >> 28) & 0x07;
  int minor = (v >> 24) & 0x0f;
  int bugFix = (v >> 16) & 0xff;

  int bcdTop = (bugFix >> 4) & 0xf;
  int bcdBottom = bugFix & 0xf;
  bugFix = bcdTop * 10 + bcdBottom;

  int build = v & 0xffff;
  return PeripheralVersion(major, minor, bugFix, build);
}
